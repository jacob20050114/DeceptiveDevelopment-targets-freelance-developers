\section{第一階段：Social Engineering 與 Initial Access}
% ============================================================

\begin{frame}{建立假身分與接觸目標}
  \begin{columns}[T]
    \begin{column}{0.5\textwidth}
      \textbf{偽裝手法}
      \begin{itemize}
        \item 複製真實人物個人檔案
        \item 偽裝成公司 Recruiter
        \item 使用成熟帳號（含被 Compromise 的真實帳號）增加可信度
      \end{itemize}
    \end{column}
    \begin{column}{0.5\textwidth}
      \textbf{活動平台}
      \begin{itemize}
        \item LinkedIn
        \item Upwork
        \item Freelancer.com
        \item We Work Remotely
        \item Moonlight
        \item Crypto Jobs List
      \end{itemize}
    \end{column}
  \end{columns}
  \vspace{0.5cm}
  \begin{alertblock}{帳號經營策略演進}
    初期使用全新帳號直接發送惡意連結 $\rightarrow$ 後期改用大量追蹤者的成熟帳號
  \end{alertblock}
\end{frame}

\begin{frame}{Social Engineering 誘騙方式}
  \begin{block}{誘餌類型}
    \begin{itemize}
      \item 「招聘程式挑戰」— 要求完成程式測試任務
      \item 「幫忙修 bug 可獲報酬」— 以報酬引誘
    \end{itemize}
  \end{block}

  \textbf{Trojanized 專案交付方式：}
  \begin{enumerate}
    \item 直接檔案傳輸
    \item GitHub / GitLab / Bitbucket Private Repository 連結
      % \begin{itemize}
      %   \item 要求受害者提供帳號 ID 或 email 以授權存取
      %   \item 藉此隱藏惡意活動的可見性
      % \end{itemize}
  \end{enumerate}

  \textbf{Trojanized 專案四大類型：}
  \begin{enumerate}
    \item Hiring Challenges
    \item Cryptocurrency Projects
    \item 含 Blockchain 功能的遊戲
    \item 含 Blockchain／Cryptocurrency 功能的賭博應用
  \end{enumerate}
\end{frame}

\begin{frame}{攻擊手段}
  \begin{block}{手段一、隱藏 Malicious Code}
    將惡意 JavaScript 程式碼放在專案中一個\textbf{看似無關的 Backend Component}裡，
    在一段\textbf{超長註解之後}以單行附加，使其被推到畫面之外 (off-screen)
  \end{block}

  \begin{itemize}
    \item GitHub 預設不啟用 word wrap
    \item 即使在線上檢視程式碼也很難發現
    \item Repository 通常複製現有 Open Source 專案或 demo，僅加入 Malicious Code 並修改 README
    \item 部分專案冒充合法公司，名稱加上 LLC、Ag、Inc 後綴
  \end{itemize}

  \vspace{0.3cm}
  \begin{block}{手段二、替代 Compromise Vector}
    邀請參加線上視訊面試，提供假視訊會議網站（如 mirotalk[.]net），
    受害者下載的「會議軟體」即為第一階段 Malware 的 Native 版本
  \end{block}
\end{frame}

\section{第二階段：BeaverTail Malware}
% ============================================================

\begin{frame}{BeaverTail — 第一階段 Infostealer / Downloader}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{block}{Target Systems}
        \begin{itemize}
          \item Windows（所有版本）
          \item Linux
          \item macOS（含 Darwin 偵測）
        \end{itemize}
      \end{block}
      \begin{block}{Target Information}
        \begin{itemize}
          \item Cryptocurrency Wallet Browser Extension 資料
          \item 瀏覽器儲存的 Login Credentials
          \item Solana 金鑰（\texttt{id.json}）
          \item macOS Keychain / Linux Keyrings
          \item Firefox 密碼資料庫
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{block}{JavaScript 版}
        \begin{itemize}
          \item 直接嵌入 Trojanized 專案
          \item IP 與 port 以 base64 編碼，拆成三段並交換順序，前方加假字元阻礙解碼
        \end{itemize}
      \end{block}
      \begin{block}{Native 版 (C++ / Qt)}
        \begin{itemize}
          \item 偽裝成視訊會議軟體
          \item 透過假會議網站散布
          \item C\&C 位址以 Plaintext 儲存
        \end{itemize}
      \end{block}
    \end{column}
  \end{columns}

  \vspace{0.2cm}
  \textbf{C\&C 連線：}Hardcoded IP，常用 port \texttt{1224} / \texttt{1244}
  \hfill \textbf{每個 Sample 含唯一 Victim ID}
\end{frame}

\begin{frame}{BeaverTail — Data Stealing 能力}
  \begin{block}{Target Extensions (Hardcoded)}
    收集 Chrome、Edge、Opera、Brave 中的 Wallet Extension \texttt{.ldb} 和 \texttt{.log} 檔案：
    \begin{itemize}
      \item MetaMask、BNB Chain Wallet、Coinbase Wallet、TronLink
      \item Phantom、Ronin Wallet、Coin98 Wallet、Crypto.com Wallet
      \item \textbf{新增：} Kaia Wallet、Rabby Wallet、Argent X、Exodus Web3 Wallet
    \end{itemize}
  \end{block}

  \begin{block}{竊取 Solana Keys}
    搜尋 \texttt{\~{}/.config/solana/id.json}
  \end{block}

  \begin{block}{竊取瀏覽器 Login Credentials}
    \begin{itemize}
      \item \textbf{macOS：} \texttt{/Library/Keychains/login.keychain}
      \item \textbf{Linux：} \texttt{\~{}/.local/share/keyrings/}
      \item \textbf{Firefox：} \texttt{key3.db}、\texttt{key4.db}、\texttt{logins.json}
    \end{itemize}
  \end{block}
\end{frame}

\section{補充：Browser Wallet Extension 架構}
% ============================================================

\begin{frame}{Browser Wallet Extension 內部架構}
  \begin{block}{三層隔離執行環境}
    \begin{enumerate}
      \item \textbf{DApp 分頁}（網頁 JS 環境）\\
            透過 \texttt{window.ethereum}（Provider，EIP-1193）與 Wallet 溝通，\textbf{無法直接存取私鑰}
      \item \textbf{Content Script}（注入腳本）\\
            與 DApp 共享 DOM 但擁有獨立 JS scope，作為 DApp 與 Extension 的中介，透過 \texttt{chrome.runtime.sendMessage()} 轉發請求
      \item \textbf{Extension Process}（完全隔離）
            \begin{itemize}
              \item \textbf{Background Service Worker}：核心邏輯處理器，負責金鑰管理、RPC 廣播、權限管理
              \item \textbf{Popup UI}：使用者確認視窗（連接授權、交易詳情）
              \item \textbf{Keystore}：私鑰以 AES-256-GCM 加密存放於本機磁碟
            \end{itemize}
    \end{enumerate}
  \end{block}
\end{frame}

\begin{frame}{金鑰系統與安全邊界}
  \begin{block}{助記詞 → 私鑰 → 地址 的衍生過程}
    \begin{itemize}
      \item 助記詞（12/24 個英文單字）$\xrightarrow{\text{BIP-39}}$ 512-bit Seed
      \item Seed $\xrightarrow{\text{BIP-32/44}}$ 私鑰（32 bytes，secp256k1）
      \item 私鑰 $\xrightarrow{\text{橢圓曲線}}$ 公鑰 $\xrightarrow{\text{Keccak-256}}$ 地址（0x 開頭，20 bytes）
      \item Solana 使用 Ed25519，公鑰直接等於錢包地址（Base58）
    \end{itemize}
  \end{block}

  % \begin{block}{私鑰的本地加密儲存}
  %   \begin{itemize}
  %     \item 使用者密碼 $\xrightarrow{\text{PBKDF2/scrypt}}$ Derived Key $\xrightarrow{\text{AES-256-GCM}}$ 加密後的 Keystore JSON
  %     \item MetaMask Vault 儲存於 \texttt{chrome.storage.local}
  %     \item \textbf{私鑰永遠不離開本機}，簽名在本地完成，僅送出已簽名的交易
  %   \end{itemize}
  % \end{block}

  \begin{alertblock}{攻擊者為何竊取 \texttt{.ldb} / \texttt{.log} 檔案？}
    這些檔案位於 Extension 的 Local Storage 目錄，可能包含加密後的 Vault 資料。若攻擊者取得這些檔案，可離線嘗試暴力破解使用者密碼以還原私鑰。
  \end{alertblock}
\end{frame}

\section{第三階段：InvisibleFerret（四大 Modules）}
% ============================================================

\begin{frame}{InvisibleFerret — 概述與部署流程}
  \begin{block}{Modular Python Backdoor，由四大模組組成}
    \begin{enumerate}
      \item \textbf{Main Module}（\texttt{.npl}）— Loader，負責下載與執行其他模組
      \item \textbf{Payload Module}（\texttt{pay}）— TCP Backdoor、Keylogger、Clipboard Stealer
      \item \textbf{Browser Module}（\texttt{bow}）— 竊取瀏覽器密碼與信用卡
      \item \textbf{AnyDesk Module}（\texttt{adc}）— Persistence Remote Access
    \end{enumerate}
  \end{block}

  \textbf{部署流程：}
  \begin{enumerate}
    \item BeaverTail 從 C\&C 下載 \texttt{p2.zip}（獨立 Python 環境）
    \item 從端點 \texttt{/client/<campaign\_ID>} 下載 Main 模組 \texttt{.npl}
    \item 儲存於使用者家目錄，使用下載的 Python 環境執行
  \end{enumerate}

  \begin{alertblock}{Module Decryption 機制}
    所有模組均以 \textbf{XOR Encryption + base64 Encoding}封裝，前 4 bytes 為 XOR Key，透過 \texttt{exec()} 執行
  \end{alertblock}
\end{frame}

\begin{frame}{模組一：Main 模組（\texttt{.npl}）— sType 與 C\&C}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \textbf{自動執行功能：}
      \begin{itemize}
        \item System Fingerprinting（UUID、OS、PC Name、Username、System Version）
        \item 取得 Local IP 與 Public IP
        % \item 透過 \texttt{ip-api.com} 取得 Geolocation
        \item 上傳至 C\&C 的 \texttt{/keys} Endpoint
      \end{itemize}

      \textbf{下載子模組：}
      \begin{itemize}
        \item \texttt{/payload/<ID>} $\rightarrow$ \texttt{.n2/pay}
        \item \texttt{/brow/<ID>} $\rightarrow$ \texttt{.n2/bow}（非 macOS）
      \end{itemize}

      C\&C 位址：
      \begin{itemize}
        \item 以 base64 Encoding 後拆成兩半並交換順序，增加 Decoding 難度
      \end{itemize}
    \end{column}
    \begin{column}{0.46\textwidth}
      \begin{block}{sType 識別碼}
        \begin{itemize}
          \item 每個模組內嵌 \texttt{sType} 變數
          \item 作為 victim/campaign 識別符
          \item 下載模組時附加於 URL 末端
          \item C\&C 伺服器將其寫入腳本
          \item 資料外傳時放入 POST 的 \texttt{type} 欄位
          \item 例：\texttt{5Team9}、\texttt{7tEaM:}
        \end{itemize}
      \end{block}
      % \vspace{0.05cm}     

      % \begin{alertblock}{注意}
      %   除 System Fingerprinting 自動採集外，所有 Backdoor 功能均需 C\&C 手動下達指令才會執行
      % \end{alertblock}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Main 模組 — C\&C 通訊架構}
  \begin{block}{Main Module 與 C\&C 的通訊重點}
    \begin{itemize}
      \item 啟動後自動透過 HTTP POST 將 System Fingerprint 上傳至 C\&C 的 \texttt{/keys} 端點，依 C\&C 回應下載子模組（Payload / Browser / AnyDesk / Clipboard）
      \item \textbf{Payload Module 的 C\&C 位址可能與 Main Module 不同}，藉此將 Backdoor Traffic 與 Initial Deployment Traffic 分離，降低被偵測的風險
      \item 所有 Client 與 C\&C 的通訊（除下載 Python 環境外）均帶有 Campaign ID，確保 Operator 可追蹤每位 Victim
    \end{itemize}
  \end{block}
\end{frame}

\begin{frame}{Main 模組 — C\&C 通訊架構}
  \begin{block}{通訊格式}
    \begin{itemize}
      \item \textbf{Module Download：}GET Request，ID 附加於 URL 末端\\
            格式：\texttt{<C\&C\_IP>:<port>/<module>/<sType>/<gType>}
      \item \textbf{Data Exfiltration：}HTTP POST，ID 放入 \texttt{type} 欄位
      \item \textbf{Backdoor 通訊：}JSON over TCP Socket，含 \texttt{code}（Command ID）與 \texttt{args}（Data）兩欄位
    \end{itemize}
  \end{block}
\end{frame}

\begin{frame}{模組二：Payload Module（\texttt{pay}）— Backdoor Commands}
  \textbf{TCP Backdoor}（常用 port 1245，亦見 80、2245、3001、5000）

  \vspace{0.2cm}
  \begin{small}
  \begin{tabular}{ll}
    \toprule
    \textbf{Command} & \textbf{Function} \\
    \midrule
    \texttt{ssh\_cmd} & Remove Compromise 痕跡 \\
    \texttt{ssh\_obj} & 透過 System Shell 執行任意指令 \\
    \texttt{ssh\_clip} & Exfiltrate Keylogger 與 Clipboard 內容 \\
    \texttt{ssh\_run} & 下載並執行 Browser Module \\
    \texttt{ssh\_upload} & 透過 FTP Exfiltrate 檔案（XOR Encrypted） \\
    \texttt{ssh\_kill} & 終止 Chrome / Brave Process \\
    \texttt{ssh\_any} & 下載並安裝 AnyDesk Module \\
    \texttt{ssh\_env} & Upload 使用者 Home Directory 等大量資料 \\
    \bottomrule
  \end{tabular}
  \end{small}

  \vspace{0.3cm}
  \begin{itemize}
    \item \textbf{Keylogger}：僅 Windows，使用 \texttt{pyWinHook}
    \item \textbf{Clipboard Stealer}：使用 \texttt{pyperclip} 監控內容變化
    \item \textbf{FTP Exfiltration}：檔案以 XOR Encryption（Key \texttt{G01d\^{}6@(}），僅上傳 < 20MB
  \end{itemize}
\end{frame}

\begin{frame}{補充：Clipboard}
  \begin{block}{Clipboard（剪貼簿）}
    \begin{itemize}
      \item 作業系統提供的暫存區域，用於在不同應用程式之間複製與貼上資料
      \item 當使用者執行「複製」時，資料（文字、圖片等）會暫存於 Clipboard
      \item 攻擊者利用 \texttt{pyperclip} 持續監控 Clipboard 內容變化，可竊取使用者複製的密碼、Cryptocurrency Wallet Address、敏感文字等
      \item 危險：使用者習慣以複製貼上方式輸入密碼或轉帳地址
    \end{itemize}
  \end{block}

\end{frame}

\begin{frame}{補充：FTP}
  \begin{block}{FTP（File Transfer Protocol）}
    \begin{itemize}
      \item 一種用於在網路上傳輸檔案的標準通訊協定（RFC 959）
      \item 採用 Client-Server 架構，Client 連線至 FTP Server 進行檔案上傳或下載
      \item 在本攻擊中，Payload Module 透過 FTP 將竊取的檔案（XOR Encrypted）上傳至攻擊者控制的 Server
      \item 攻擊者選擇 FTP 而非 HTTP 的原因：適合大量檔案批次傳輸，且較不易被 Web Application Firewall 攔截
    \end{itemize}
  \end{block}
\end{frame}

\begin{frame}{模組三：Browser Module（\texttt{bow}）— Browser Data Stealing}
  \textbf{Target Browsers：}Chrome、Brave、Opera、Yandex、Edge

  \vspace{0.3cm}
  \begin{block}{Stolen Data}
    \begin{itemize}
      \item \texttt{LoginData.db} — User Login Information（Credentials）
      \item \texttt{webdata.db} — Saved Credit Card Information
    \end{itemize}
  \end{block}

  \vspace{0.3cm}
  \textbf{AES Decryption Key 取得方式：}
  \begin{itemize}
    \item \textbf{Windows：}從瀏覽器 \texttt{Local State} 檔案取得
    \item \textbf{Linux：}透過 \texttt{secretstorage} Package 取得
    \item \textbf{macOS：}透過 \texttt{security} Utility 取得
  \end{itemize}

  \vspace{0.3cm}
  資料透過 HTTP POST 上傳至 C\&C 的 \texttt{/keys} Endpoint
\end{frame}

\begin{frame}{模組四：AnyDesk Module（\texttt{adc}）— Persistence}
  \begin{alertblock}{Persistence Mechanism}
    利用合法 Remote Desktop 軟體 AnyDesk 建立 Persistent Backdoor
  \end{alertblock}

  \textbf{部署流程：}
  \begin{enumerate}
    \item 檢查 \texttt{C:/Program Files (x86)/AnyDesk/AnyDesk.exe} 是否存在
    \item 若無，從 C\&C 下載 \texttt{anydesk.exe} 至使用者 Home Directory
    \item 將 Hardcoded 的 Password Hash、Salt、Token Salt 寫入設定檔
    \item 確保攻擊者可隨時 Remote Access
  \end{enumerate}

  \vspace{0.3cm}
  \textbf{備用方案：}
  \begin{itemize}
    \item 若直接修改設定失敗，建立 \texttt{conf.ps1} PowerShell Script 修改設定
    \item 終止 AnyDesk Process 後重新啟動以載入新設定
    \item 模組最後呼叫 \texttt{os.remove} 刪除自身（Anti-forensics）
  \end{itemize}
\end{frame}

\section{後續版本更新與演進}
% ============================================================

\begin{frame}{2024 年 8 月後版本更新（一）— 架構與識別碼}
  \begin{block}{架構變化}
    \begin{itemize}
      \item InvisibleFerret 合併為\textbf{單一 Script}（不再分獨立模組）
      \item 專案中的程式碼改為 \textbf{Loader}，從遠端下載實際 Payload
    \end{itemize}
  \end{block}
