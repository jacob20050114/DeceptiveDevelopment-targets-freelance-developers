\begin{frame}{社交工程誘騙方式}
  \begin{block}{誘餌類型}

  \textbf{木馬化專案交付方式：}
  \begin{enumerate}
    \item 直接檔案傳輸
    \item GitHub / GitLab / Bitbucket 私有倉庫連結
      \begin{itemize}
        \item 要求受害者提供帳號 ID 或 email 以授權存取
        \item 藉此隱藏惡意活動的可見性
      \end{itemize}
  \end{enumerate}

  \textbf{木馬化專案四大類型：}
  \begin{enumerate}
    \item 招聘程式挑戰 (hiring challenges)
    \item 加密貨幣專案 (cryptocurrency projects)
    \item 含區塊鏈功能的遊戲
    \item 含區塊鏈／加密貨幣功能的賭博應用
  \end{enumerate}
\end{frame}

\begin{frame}{攻擊手段}
  \begin{alertblock}{手段一、隱藏惡意程式}
    將惡意 JavaScript 程式碼放在專案中一個\textbf{看似無關的後端組件}裡，
    在一段\textbf{超長註解之後}以單行附加，使其被推到畫面之外 (off-screen)
  \end{alertblock}

  \begin{itemize}
    \item GitHub 預設不啟用 word wrap
    \item 即使在線上檢視程式碼也很難發現
    \item 倉庫通常複製現有開源專案或 demo，僅加入惡意程式碼並修改 README
    \item 部分專案冒充合法公司，名稱加上 LLC、Ag、Inc 後綴
  \end{itemize}

  \vspace{0.3cm}
  \begin{block}{手段二、替代感染途徑}
    邀請參加線上視訊面試，提供假視訊會議網站（如 mirotalk[.]net），
    受害者下載的「會議軟體」即為第一階段惡意程式的原生版本
  \end{block}
\end{frame}

\section{第三階段：InvisibleFerret（四大模組）}
% ============================================================

\begin{frame}{InvisibleFerret — 概述與部署流程}
  \begin{block}{模組化 Python 後門，由四大模組組成}
    \begin{enumerate}
      \item \textbf{Main 模組}（\texttt{.npl}）— loader，負責下載與執行其他模組
      \item \textbf{Payload 模組}（\texttt{pay}）— TCP 後門、鍵盤側錄、剪貼簿竊取
      \item \textbf{Browser 模組}（\texttt{bow}）— 竊取瀏覽器密碼與信用卡
      \item \textbf{AnyDesk 模組}（\texttt{adc}）— 持久化遠端存取
    \end{enumerate}
  \end{block}

  \textbf{部署流程：}
  \begin{enumerate}
    \item BeaverTail 從 C\&C 下載 \texttt{p2.zip}（獨立 Python 環境）
    \item 從端點 \texttt{/client/<campaign\_ID>} 下載 Main 模組 \texttt{.npl}
    \item 儲存於使用者家目錄，使用下載的 Python 環境執行
  \end{enumerate}

  \begin{alertblock}{模組解密機制}
    所有模組均以 \textbf{XOR 加密 + base64 編碼}封裝，前 4 bytes 為 XOR 金鑰，透過 \texttt{exec()} 執行
  \end{alertblock}
\end{frame}

\begin{frame}{模組一：Main 模組（\texttt{.npl}）— sType 與 C\&C}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \textbf{自動執行功能：}
      \begin{itemize}
        \item 系統指紋採集（UUID、OS、PC 名稱、使用者名稱、系統版本）
        \item 取得本機 IP 與公開 IP
        \item 透過 \texttt{ip-api.com} 取得地理位置
        \item 上傳至 C\&C 的 \texttt{/keys} 端點
      \end{itemize}

      \vspace{0.2cm}
      \textbf{下載子模組：}
      \begin{itemize}
        \item \texttt{/payload/<ID>} $\rightarrow$ \texttt{.n2/pay}
        \item \texttt{/brow/<ID>} $\rightarrow$ \texttt{.n2/bow}（非 macOS）
      \end{itemize}

      \vspace{0.2cm}
      \textbf{C\&C 位址：}
      \begin{itemize}
        \item 以 base64 編碼後拆成兩半並交換順序，增加解碼難度
      \end{itemize}
    \end{column}
    \begin{column}{0.46\textwidth}
      \begin{block}{sType 識別碼}
        \begin{itemize}
          \item 每個模組內嵌 \texttt{sType} 變數
          \item 作為 victim/campaign 識別符
          \item 下載模組時附加於 URL 末端
          \item C\&C 伺服器將其寫入腳本
          \item 資料外傳時放入 POST 的 \texttt{type} 欄位
          \item 例：\texttt{5Team9}、\texttt{7tEaM:}
        \end{itemize}
      \end{block}
      % \vspace{0.05cm}     

      \begin{alertblock}{注意}
        除系統指紋自動採集外，所有後門功能均需操作者手動下達指令才會執行
      \end{alertblock}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Main 模組 — C\&C 通訊架構}
  \begin{block}{C\&C 位址特性}
    \begin{itemize}
      \item Main 模組的 C\&C 位址大多與前段 BeaverTail 相同
      \item \textbf{Payload 模組的 C\&C 位址可能不同}，用以分離可疑的後門流量與初始部署流量
      \item 所有客戶端到 C\&C 的通訊（除下載 Python 環境外）均包含 campaign ID
    \end{itemize}
  \end{block}

  \begin{block}{通訊格式}
    \begin{itemize}
      \item \textbf{下載模組：}GET 請求，ID 附加於 URL 末端\\
            格式：\texttt{<C\&C\_IP>:<port>/<module>/<sType>/<gType>}
      \item \textbf{資料上傳：}HTTP POST，ID 放入 \texttt{type} 欄位
      \item \textbf{後門通訊：}JSON over TCP socket，含 \texttt{code}（指令 ID）與 \texttt{args}（資料）兩欄位
    \end{itemize}
  \end{block}
\end{frame}

\begin{frame}{模組二：Payload 模組（\texttt{pay}）— 後門指令}
  \textbf{TCP 後門}（常用 port 1245，亦見 80、2245、3001、5000）

  \vspace{0.2cm}
  \begin{small}
  \begin{tabular}{ll}
    \toprule
    \textbf{指令} & \textbf{功能} \\
    \midrule
    \texttt{ssh\_cmd} & 終止操作並移除感染痕跡 \\
    \texttt{ssh\_obj} & 透過系統 shell 執行任意指令 \\
    \texttt{ssh\_clip} & 回傳鍵盤側錄與剪貼簿內容 \\
    \texttt{ssh\_run} & 下載並執行 browser 模組 \\
    \texttt{ssh\_upload} & 透過 FTP 外傳檔案（XOR 加密） \\
    \texttt{ssh\_kill} & 終止 Chrome / Brave 進程 \\
    \texttt{ssh\_any} & 下載並安裝 AnyDesk 模組 \\
    \texttt{ssh\_env} & 上傳使用者家目錄等大量資料 \\
    \bottomrule
  \end{tabular}
  \end{small}

  \vspace{0.3cm}
  \begin{itemize}
    \item \textbf{鍵盤側錄}：僅 Windows，使用 \texttt{pyWinHook}
    \item \textbf{剪貼簿竊取}：使用 \texttt{pyperclip} 監控內容變化
    \item \textbf{FTP 外傳}：檔案以 XOR 加密（金鑰 \texttt{G01d\^{}6@(}），僅上傳 < 20MB
  \end{itemize}
\end{frame}

\begin{frame}{模組三：Browser 模組（\texttt{bow}）— 瀏覽器資料竊取}
  \textbf{目標瀏覽器：}Chrome、Brave、Opera、Yandex、Edge

  \vspace{0.3cm}
  \begin{block}{竊取內容}
    \begin{itemize}
      \item \texttt{LoginData.db} — 使用者登入資訊（帳號密碼）
      \item \texttt{webdata.db} — 儲存的信用卡資訊
    \end{itemize}
  \end{block}

  \vspace{0.3cm}
  \textbf{AES 解密金鑰取得方式：}
  \begin{itemize}
    \item \textbf{Windows：}從瀏覽器 \texttt{Local State} 檔案取得
    \item \textbf{Linux：}透過 \texttt{secretstorage} 套件取得
    \item \textbf{macOS：}透過 \texttt{security} 工具取得
  \end{itemize}

  \vspace{0.3cm}
  資料透過 HTTP POST 上傳至 C\&C 的 \texttt{/keys} 端點
\end{frame}

\begin{frame}{模組四：AnyDesk 模組（\texttt{adc}）— 持久化遠端存取}
  \begin{alertblock}{持久化機制}
    利用合法遠端桌面軟體 AnyDesk 建立持久後門
  \end{alertblock}

  \textbf{部署流程：}
  \begin{enumerate}
    \item 檢查 \texttt{C:/Program Files (x86)/AnyDesk/AnyDesk.exe} 是否存在
    \item 若無，從 C\&C 下載 \texttt{anydesk.exe} 至使用者家目錄
    \item 將硬編碼的 password hash、salt、token salt 寫入設定檔
    \item 確保攻擊者可隨時遠端存取
  \end{enumerate}

  \vspace{0.3cm}
  \textbf{備用方案：}
  \begin{itemize}
    \item 若直接修改設定失敗，建立 \texttt{conf.ps1} PowerShell 腳本修改設定
    \item 終止 AnyDesk 進程後重新啟動以載入新設定
    \item 模組最後呼叫 \texttt{os.remove} 刪除自身（反鑑識）
  \end{itemize}
\end{frame}

\section{後續版本更新與演進}
% ============================================================

\begin{frame}{2024 年 8 月後版本更新（一）— 架構與識別碼}
  \begin{block}{架構變化}
    \begin{itemize}
      \item InvisibleFerret 合併為\textbf{單一腳本}（不再分獨立模組）
      \item 專案中的程式碼改為 \textbf{loader}，從遠端下載實際 payload
    \end{itemize}
  \end{block}

  \begin{block}{新增 gType 識別碼}
    \begin{itemize}
      \item 在 \texttt{sType} 之外新增 \texttt{gType} 作為次要 victim/campaign 識別符
      \item 下載模組 URL 格式變更為：\texttt{<C\&C\_IP>:<port>/<module>/<sType>/<gType>}
      \item 引入 gType 前，sType 為 base64 字串（如 \texttt{5Team9}、\texttt{7tEaM:}）
      \item 引入 gType 後，兩者多為純數字，不再使用 base64
    \end{itemize}
  \end{block}
\end{frame}

\begin{frame}{2024 年 8 月後版本更新（二）— 新增功能}
  \begin{alertblock}{新增 ssh\_zcp 指令}
    \begin{itemize}
      \item 可從 \textbf{88 個瀏覽器擴充套件}中提取資料
      \item 竊取 Atomic、Exodus 加密貨幣錢包
      \item 竊取 1Password、Electrum、WinAuth、Proxifier4、Dashlane 等密碼管理器
    \end{itemize}
  \end{alertblock}

  \begin{block}{資料外傳至 Telegram}
    使用 Telegram Bot API 將 ZIP/7z 檔案上傳至 Telegram 聊天室，同時傳至 FTP
  \end{block}
\end{frame}

\begin{frame}{2024 年 12 月 — 獨立 Clipboard Stealer}
  \begin{block}{新模組 \texttt{mlip}}
    \begin{itemize}
      \item 從端點 \texttt{/mclip/<campaign\_ID>} 下載至 \texttt{.n2/mlip}
      \item 將鍵盤側錄與剪貼簿竊取功能獨立出來
      \item \textbf{精準化監控：}僅監控 \texttt{chrome.exe} 和 \texttt{brave.exe}
      \item 資料上傳至新端點 \texttt{/api/clip}
    \end{itemize}
  \end{block}

  \vspace{0.3cm}
  \textbf{演進趨勢：}
  \begin{itemize}
    \item 從廣泛監控 $\rightarrow$ 精準化目標監控
    \item 模組化 $\rightarrow$ 單一腳本 $\rightarrow$ 再次模組化（功能分離）
    \item 持續擴大竊取範圍（88 個擴充套件 + 密碼管理器）
  \end{itemize}
\end{frame}
